GRAPH NOTES: Added for SpatialAMR.

graphrel

              :polarity-of (s / symmetric-01
                :ARG1 (t2 / thing

                    :ARG1 (s2 / string-entity :value "y = |x| + 3")
                    :ARG2 (c / cartesian-framework-91)))
                :ARG4 (o / or
                  :op1 (o2 / origin
                    :ARG5-of c)
                  :op2 (x / x-axis
                    :ARG2-of c)
                  :op3 (y2 / y-axis
                    :ARG3-of c)))))
